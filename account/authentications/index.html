PUT /authentications

Headers:

{
    "Authorization": "urAccesToken"
}

Body:

{
    "refreshToken": "urRefreshToken"
}

Response:

{
    "status": "success",
    "message": "Access Token berhasil diperbarui",
    "data": {
        "accessToken": "urNewAccessToken"
    }
}
