POST /pets

Headers:

{
    "Authorization": "urAccesToken"
}

Body: multiform

Parameter Type Required
name string Yes
species string Yes
breed string No
description string No
imageUpload file(image) No

Response:

{
    "status": "success",
    "message": "Pet berhasil ditambahkan",
    "data": {
        "petId": "urPetId"
    }
}
